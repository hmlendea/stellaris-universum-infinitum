    time = 300

    is_for_colonizeable = yes
    category = deposit_cat_blockers

    resource = {
        category = deposit_blockers
        cost = {
            energy = 500
        }
    }

    potential = {
        is_dry = yes
    }

    planet_modifier = {
        planet_max_districts_add = -1
    }

    all_blocker_swap_types = yes

    drop_weight = {
        weight = @blocker_medium
        modifier = {
            factor = 0.5
            planet_size < 15
        }
        modifier = {
            factor = 3
            is_planet_class = pc_nuked
        }
        modifier = {
            factor = 2
            is_planet_class = pc_alpine
        }
        modifier = {
            factor = 0
            num_free_districts = {
                type = district_city
                value < 2
            }
        }
    }

    prerequisites = { "tech_d_ui_ashland"_blocker" }
}

# Giant Earthworms
d_ui_giant_earthworms_blocker = {
    time = 100

    is_for_colonizeable = yes
    category = deposit_cat_blockers
    
    resources = {
        cost = {
            energy = 200
        }
    }

    potential = {
        AND = {
            is_dry = yes
            planet_size >= 10
        }
    }

    planet_modifier = {
        planet_max_districts_add = -2
    }

    on_cleared = {
        hidden_effect = {
            if = {
                limit = {
                    exists = owner
                    owner = {
                        has_civic = civic_hive_devouring_swarm
                    }
                }
                owner = {
                    add_resource = {
                        food = 200
                    }
                }
            }
        }
    }

    drop_weight = {
        weight = @blocker_low
        modifier = {
            factor = 0.5
            planet_size < 15
        }
        modifier = {
            factor = 0
            num_free_districts = {
                type = district_city
                value < 2
            }
        }
    }

    prerequisites = { "tech_d_ui_giant_earthworms"_blocker" }
}
