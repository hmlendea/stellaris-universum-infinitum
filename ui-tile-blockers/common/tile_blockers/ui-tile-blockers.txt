# Ashland
tb_ui_ashland = {
    time = 300
    cost = {
        energy = 400
    }

    spawn_chance = {
        modifier = {
            add = 12
            planet = {
                is_colonizable = yes
                AND = {
                    NOT = { is_planet_class = pc_nuked }
                    NOT = { is_planet_class = pc_alpine }
                }
            }
        }
        modifier = {
            add = 24
            planet = {
                is_colonizable = yes
                s_planet_class = pc_nuked
            }
        }
        modifier = {
            add = 6
            planet = {
                is_colonizable = yes
                is_planet_class = pc_alpine
            }
        }
        modifier = {
            add = 6
            planet = {
                any_tile = {
                    has_blocker = "tb_active_volcano"
                }
            }
        }
        modifier = {
            factor = 0.25
            planet = {
                planet_size < 15
            }
        }
    }

    prerequisites = { "tech_tb_ui_ashland" }
}
